/* Program 18:

character */

#include <ctype.h>

{
	char ch, op;
	printf("\nEnter a character : ");
	ch = getchar();

	if(isalpha(ch))
	{
		printf("\n%c is an alphabet",ch);
		if(islower(ch))
			op = toupper(ch);
		else if(isupper(ch))
			op = tolower(ch);

		printf("\nch = %c\top = %c\n",ch,op);
	}
	else if(isdigit(ch))
		printf("\n%c is a digit\n",ch);
	else
		printf("\n%c is a special character\n",ch);

		if(isalpha(text[i]))

				text[i] = text[i]+1;

		else if(isdigit(text[i]))

			if(text[i] == '9')

				text[i] = text[i]+1;

	for(i=0; i<strlen(text); i++)	// for(i=0; text[i]!='\0'; i++)

			if(text[i] == '0')

/* Program 46 : Multi-Dimensional Arrays
	Write a program to calculate the total average marks of each section of a class
and display the summary report of each section. Use 3 dimensional numeric array */

#include <conio.h>	// getch()
#define line for(k=0; k<75; k++) printf("-")

char subjects[6][15] = {"English" , "Telugu", "Hindi", "Maths", "Social", "Science" };
char ok = 'y';
void main()
{
	int marks[5][10][7];
	/*
		Max. 5 sections. Max 10 students/section. subjects 6+1 for total
	*/
	int i, j, k, sz, sum , total=0;
	int capacity=0;
	float avg;
	int strength[5];	// stores each section's no.of students

	do
	{
		if(ok == 'y')
			printf("\nEnter no.of sections (Max. 5) : ");
		else
			printf("Maximum sections allowed is only 5.\nEntered value %d is not accepted.\nPlease Enter the correct value(1 to 5) to proceed : ",sz);
		scanf("%d",&sz);

		if(sz<=0 || sz>5)
			ok='n';
		else
			ok='y';
	}while(sz<=0 || sz>5);

	for(i=0; i<sz; i++)
	{
		do
		{
			if(ok == 'y')
				printf("\nEnter no.of students in section %d (max. 10 students) : ",i+1);
			else
			{
				printf("\nEnter value %d is not accepted.\nMaximum students allowed per section is only 10\n",capacity);
				printf("\nEnter the correct no.of students  in section %d to proceed : ",i+1);

			scanf("%d",&capacity);

				ok = 'n';
			else
				ok = 'y';

		}while(capacity<=0 || capacity>10);

			sum=0;

					{
						printf("\nInvalid mark. Please enter the correct mark (1 to 100)\n");
						printf("Enter marks of student %d in subject \" %s \" of section %d : ",j+1,subjects[k],i+1);
					}
					scanf("%d",&marks[i][j][k]);
					if( marks[i][j][k]<0 || marks[i][j][k]>100 )
						ok = 'n';
					else
						ok = 'y';
				}while(marks[i][j][k]<0 || marks[i][j][k]>100);
				sum = sum + marks[i][j][k];
			}

		}

	// Summary report of each section
	getch();

		total = 0;

			printf("\n");
		}

		printf("\nNo.of students present in section %d = %d",i+1,strength[i]);
		printf("\nAverage total marks = %.2f\n",avg);

		getch();
	}
	printf("\n");
}

#include <conio.h>	// getch()

{

	printf("\nEnter no.of paragraphs(max. 10) : ");

	{
		printf("\n");
		for(j=0; j<nlines[i]; j++)

	}
