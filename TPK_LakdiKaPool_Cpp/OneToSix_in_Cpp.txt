/* Program 10 : Write a C++ program to find the roots of a quadratic equation.
	ax^2 + bx + c = 0 is a quadratic equation.
	Roots : x1 = (-b + sqrt(b^2-4ac))/2a;
			x2 = (-b - sqrt(b^2-4ac))/2a;
Note: i = sqrt(-1); ==> i^2 = -1;
*/

#include <iostream.h>
#include <math.h>

void roots(float a, float b, float c, float &r1, float &r2)
{
	float disc;		// discriminant = b^2-4ac;
	disc = b*b - 4*a*c;	
	//disc = pow(b,2) - 4*a*c;

	if(disc >= 0)
	{
		r1 = (-b + sqrt(disc))/(2*a);
		r2 = (-b - sqrt(disc))/(2*a);
	}
	else
	{
		r1 = -b/(2*a);	// real part
		r2 = sqrt(4*a*c - b*b)/(2*a);	// imaginary part
	}
}

void main()
{
	float a, b, c, disc, r1, r2;

	void roots(float, float, float, float&, float&);	// function prototype

	cout << "\nQuadratic Equation : ax^2 + bx + c = 0" << endl;
	cout << "Enter the values for a, b, c : " << '\n';
	cin >> a >> b >> c;

	roots(a, b, c, r1, r2);

	//disc = b*b - 4*a*c;
	disc = pow(b,2) - 4*a*c;

	if(disc >= 0)
	{
		cout << "\nRoot1 = " << r1 << "\n";
		cout << "Root2 = " << r2 << endl;
		cout.precision(0);
		cout << "\ncondition checking : a*pow(r1,2) + b*r1 + c = " << a*pow(r1,2)+b*r1+c << endl;
		cout << "condition checking : a*r2^2 + b*r2 + c = " << a*pow(r2,2)+b*r2+c << endl;
	}
	else
	{
		cout << "\nRoot1 = " << r1 << " + " << r2 << "i" << endl;
		cout << "Root2 = " << r1 << " - " << r2 << "i" << endl;
	}
}

/* Program 12 : Write a C++ Program to find GCD(Greatest Common Divisor) and 
LCM(Least Common Multiplier) of two numbers.
	LCM of two numbers = (Product of two numbers)/(GCD of two numbers)
*/

#include <iostream.h>

int GCD(int dividend, int divisor)
{
	int rem;	// remainder
	
	rem = dividend % divisor;
	while(rem != 0)
	{
		dividend = divisor;
		divisor = rem;
		rem = dividend % divisor;
	}
	
	return divisor;
}

int LCM(int a, int b)
{
	int GCD(int, int);		// function prototype --- optional

	return ( (a*b)/GCD(a,b) );
}

void main()
{
	int GCD(int, int);	// function prototype --- optional
	int LCM(int, int);
	
	int gcd, lcm, x, y;
	
	cout << "\nEnter any two numbers whose GCD and LCM is to be found\n";
	cin >> x >> y;

	gcd = GCD(x, y);
	lcm = LCM(x, y);

	cout << "\nGCD = " << gcd << endl;
	cout << "LCM = " << lcm << '\n';
}

/* Program 13 : Function Overloading  ----- example1
		Function Overloading is nothing but having more than one function with
same name but having different signatures i.e., either no.of parameters must be
different or parameter types must be different.
		In C language, no two functions have same name but in C++, more than
one functions have same name but should have different signatures (i.e., no.of
parameters or parameter types should be different).
	Note: whenever we want string datatype, use char* as string datatype.
*/

#include <iostream.h>
#include <string.h>

int add(int a, int b)	// 1st function
{
	return (a+b);
}

double add(double a, double b, double c)	// 2nd function
{
	return (a+b+c);
}

char* add(char *a, char *b)	// 3rd function
{
	char *c = new char[strlen(a) + strlen(b) + 1];
	strcpy(c,a);
	strcat(c,b);
	return c;
}

void main()
{
	char *ans1;
	double ans2;
	int ans3, ans4;

	ans3 = add(10,20);
	ans4 = add(10.8, 20.6);	
	// 1st function called and parameters are typecasted from float to int.

	ans1 = add("palani ","kumar");
	ans2 = add(10.8,20.6,30.2);

	cout << ans1 << endl;
	cout << ans2 << endl;
	cout << ans3 << endl;
	cout << ans4 << endl;

	if(ans1 != NULL);
		delete []ans1;
}

	------

2. Generic function must be preceded by 
		template<class T>

/* Program 34 : Write a Generic function for bubble sort and for  	display 1-D array */

#include <iostream.h>
#include <string.h>

template <class T>
void bubblesort(T *a, int sz)	// void bubblesort(T a[], int sz)
{
	T temp;
	for(int i=0; i<sz-1; i++)
		for(int j=i+1; j<sz; j++)
			if(a[i] > a[j])
			{
				temp = a[i];
				a[i] = a[j];
				a[j] = temp;
			}
}

template <class T>

{
	for(int i=0; i<sz; i++)
		cout << a[i] << "\t";
	cout << endl;
}

	display(c, n);
	bubblesort(c, n);
	cout << "\nElements after sorting " << endl;
